类函数
```cpp
#include <iostream>

using namespace std;

class Box
{
public:
    char* color;
    int size;
    long box(void); //函数原型
	long boxs()		//添加成员函数
	{
		return size*size;
	};
};

int main()
{
    Box b1;
    b1.color="red";
    b1.size=20;
    cout<<"Box b1 color is "<<b1.color<<" and size is "<<b1.size<<endl;
    cout<<b1.box()<<endl;
	cout<<b1.boxs()<<endl;
    return 0;
}

long Box::box()     //该函数属于Box类，即给函数加上类名作为前缀
{
    return ++size;
}
```

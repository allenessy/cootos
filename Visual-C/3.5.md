###for循环
依次输出1-100
```cpp
#include <iostream>

using namespace std;

int main()
{
	for(int i = 1;i<=100;i++)
	{
		cout<<i<<endl;
	}
	return 0;
}
```
输出累加1-100的和
```cpp
#include <iostream>

using namespace std;

int main()
{
	int sum=0;
	for(int i = 1;i<=100;i++)
	{
		sum+=i;
	}
	cout<<sum<<endl;
	return 0;
}
```
无穷for循环
```cpp
#include <iostream>
#include <iomanip>

using namespace std;

int main()
{
	int i = 1;
	for(;;)
	{
		cout<<i<<endl;
		i++;
		if(i>100)
			break;		//跳出循环
	}
	return 0;

}
```
continue语句
```cpp
#include <iostream>
#include <iomanip>

using namespace std;

int main()
{
	for(int i = 1;i<=10;i++)
	{
		if((i%2)!=1)
			continue;		//跳过该次循环
		cout<<i<<endl;
	}
	return 0;

}
```

###hello world MFC程序
```cpp
#include <afxwin.h>

class CHelloApp :public CWinApp
{
public:
	virtual BOOL InitInstance();
};

CHelloApp HelloApp;

class CHelloWindow :public CFrameWnd
{
	CStatic* cs;
public:
	CHelloWindow();
};

BOOL CHelloApp::InitInstance()
{
	m_pMainWnd = new CHelloWindow();
	m_pMainWnd->ShowWindow(m_nCmdShow);
	m_pMainWnd->UpdateWindow();
	return TRUE;
}

CHelloWindow::CHelloWindow()
{
	Create(NULL,
		_T("vc++程序"),
		WS_OVERLAPPEDWINDOW,
		CRect(0, 0, 200, 200)
		);

	cs = new CStatic();
	cs->Create(_T("您好，世界"),
		WS_CHILD | WS_VISIBLE | SS_CENTER,
		CRect(50, 80, 150, 150),
		this
		);
}
```

class CTestApp :public CWinApp

public:
	virtual BOOL InitInstance();

CTestApp TestApp;

{
	CStatic* cs;
public:
	CTestWindow();
};

BOOL CTestApp::InitInstance()
{
	m_pMainWnd = new CTestWindow();
	m_pMainWnd->ShowWindow(m_nCmdShow);
	m_pMainWnd->UpdateWindow();
	return TRUE;
}

CTestWindow::CTestWindow()
{
	CRect r;
	
	Create(NULL,
		_T("CStatic Tests"),
		WS_OVERLAPPEDWINDOW,
		CRect(0, 0, 200, 200)
		);

	GetClientRect(&r);
	r.InflateRect(-20, -20);

	cs = new CStatic();
	cs->Create(_T("hello world"),
		WS_CHILD | WS_VISIBLE | WS_BORDER | SS_CENTER,
		r,
		this
		);

}

#include <afxwin.h>

class CTestApp :public CWinApp
{
public:
	virtual BOOL InitInstance();
};

CTestApp TestApp;

	CStatic* cs;

	CTestWindow();
};

{
	m_pMainWnd = new CTestWindow();
	m_pMainWnd->ShowWindow(m_nCmdShow);
	m_pMainWnd->UpdateWindow();
	return TRUE;
}

CTestWindow::CTestWindow()
{
	CRect r;
	
	Create(NULL,
		_T("标题"),

		CRect(0, 0, 500, 200)
		);

	GetClientRect(&r);
	r.InflateRect(-20, -20);

	cs = new CStatic();

				  come to the aid of their country"),
